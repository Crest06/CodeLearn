JavaScript Basics

JavaScript is a programming language used to make web pages interactive. Below are some examples of JavaScript code:

1. Hello, World!


console.log("Hello, World!");

2. Variables and Data Types


let name = "Alice";
let age = 25;
let isStudent = true;

console.log(name, age, isStudent);

3. Functions


function greet(name) {
  return `Hello, ${name}!`;
}

console.log(greet("Alice"));

4. Arrays


let fruits = ["apple", "banana", "cherry"];
console.log(fruits[0]); // Access first element
fruits.push("orange"); // Add an element
console.log(fruits);

5. Conditional Statements


let age = 18;

if (age >= 18) {
  console.log("You are an adult.");
} else {
  console.log("You are a minor.");
}

6. Loops


for (let i = 0; i < 5; i++) {
  console.log(i);
}

let fruits = ["apple", "banana", "cherry"];
for (let fruit of fruits) {
  console.log(fruit);
}

7. DOM Manipulation


// Change the text of an element
document.getElementById("demo").innerText = "Hello, JavaScript!";

// Add an event listener
document.querySelector("button").addEventListener("click", () => {
  alert("Button clicked!");
});
